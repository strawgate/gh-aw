<script type="module">{`
  const SAMPLE_WORKFLOW = [
    '---',
    'name: hello-world',
    'description: A simple hello world workflow',
    'on:',
    '  workflow_dispatch:',
    'engine: copilot',
    '---',
    '',
    '# Mission',
    '',
    'Say hello to the world! Check the current date and time, and greet the user warmly.',
    '',
  ].join('\\n');

  // ---- DOM references ----
  const root = document.getElementById('live-editor-root');
  const loadingEl = document.getElementById('le-loading');
  const containerEl = document.getElementById('le-container');
  const inputEl = document.getElementById('le-input');
  const outputCodeEl = document.getElementById('le-output-code');
  const errorEl = document.getElementById('le-error');
  const warningsEl = document.getElementById('le-warnings');
  const compileBtn = document.getElementById('le-compile-btn');
  const copyBtn = document.getElementById('le-copy-btn');
  const autoToggle = document.getElementById('le-auto-toggle');
  const statusEl = document.getElementById('le-status');
  const lineNumbersEl = document.getElementById('le-line-numbers');

  let compiler = null;
  let debounceTimer = null;
  let lastYaml = '';

  // ---- Line numbers ----
  function updateLineNumbers() {
    const lines = inputEl.value.split('\\n');
    const count = lines.length;
    let html = '';
    for (let i = 1; i <= count; i++) {
      html += i + '\\n';
    }
    lineNumbersEl.textContent = html;
  }

  // Sync scroll between textarea and line numbers
  function syncScroll() {
    lineNumbersEl.scrollTop = inputEl.scrollTop;
  }

  // ---- Tab support ----
  function handleTab(e) {
    if (e.key === 'Tab') {
      e.preventDefault();
      const start = inputEl.selectionStart;
      const end = inputEl.selectionEnd;
      const value = inputEl.value;
      inputEl.value = value.substring(0, start) + '  ' + value.substring(end);
      inputEl.selectionStart = inputEl.selectionEnd = start + 2;
      inputEl.dispatchEvent(new Event('input'));
    }
  }

  // ---- Compile ----
  async function compile() {
    if (!compiler) return;

    const md = inputEl.value;
    if (!md.trim()) {
      outputCodeEl.textContent = '';
      errorEl.style.display = 'none';
      warningsEl.style.display = 'none';
      setStatus('', '');
      lastYaml = '';
      return;
    }

    setStatus('Compiling...', 'le-status-compiling');
    compileBtn.disabled = true;

    try {
      const result = await compiler.compile(md);

      if (result.error) {
        errorEl.textContent = result.error;
        errorEl.style.display = 'block';
        outputCodeEl.textContent = '';
        lastYaml = '';
        setStatus('Error', 'le-status-error');
      } else {
        errorEl.style.display = 'none';
        outputCodeEl.textContent = result.yaml || '';
        lastYaml = result.yaml || '';
        setStatus('Compiled', 'le-status-success');

        // Clear "Compiled" status after a moment
        setTimeout(() => {
          if (statusEl.textContent === 'Compiled') {
            setStatus('', '');
          }
        }, 2000);
      }

      if (result.warnings && result.warnings.length > 0) {
        warningsEl.textContent = result.warnings.join('\\n');
        warningsEl.style.display = 'block';
      } else {
        warningsEl.style.display = 'none';
      }
    } catch (err) {
      errorEl.textContent = 'Compilation error: ' + (err.message || String(err));
      errorEl.style.display = 'block';
      lastYaml = '';
      setStatus('Error', 'le-status-error');
    } finally {
      compileBtn.disabled = false;
    }
  }

  function setStatus(text, className) {
    statusEl.textContent = text;
    statusEl.className = 'le-status' + (className ? ' ' + className : '');
  }

  // ---- Debounced auto-compile ----
  function scheduleCompile() {
    if (!autoToggle.checked) return;
    clearTimeout(debounceTimer);
    debounceTimer = setTimeout(compile, 500);
  }

  // ---- Copy button ----
  async function copyYaml() {
    if (!lastYaml) return;
    try {
      await navigator.clipboard.writeText(lastYaml);
      const orig = copyBtn.textContent;
      copyBtn.textContent = 'Copied!';
      setTimeout(() => { copyBtn.textContent = orig; }, 1500);
    } catch (e) {
      // Fallback for older browsers
      const ta = document.createElement('textarea');
      ta.value = lastYaml;
      ta.style.position = 'fixed';
      ta.style.left = '-9999px';
      document.body.appendChild(ta);
      ta.select();
      document.execCommand('copy');
      document.body.removeChild(ta);
      const orig = copyBtn.textContent;
      copyBtn.textContent = 'Copied!';
      setTimeout(() => { copyBtn.textContent = orig; }, 1500);
    }
  }

  // ---- Keyboard shortcuts ----
  function handleKeyboard(e) {
    if ((e.ctrlKey || e.metaKey) && e.key === 'Enter') {
      e.preventDefault();
      compile();
    }
  }

  // ---- Wire up events ----
  inputEl.addEventListener('input', () => {
    updateLineNumbers();
    scheduleCompile();
  });
  inputEl.addEventListener('scroll', syncScroll);
  inputEl.addEventListener('keydown', handleTab);
  inputEl.addEventListener('keydown', handleKeyboard);
  compileBtn.addEventListener('click', compile);
  copyBtn.addEventListener('click', copyYaml);

  // ---- Initialize ----
  async function init() {
    try {
      const { createWorkerCompiler } = await import('/gh-aw/wasm/compiler-loader.js');
      compiler = createWorkerCompiler();
      await compiler.ready;

      // Show the editor
      loadingEl.style.display = 'none';
      containerEl.style.display = 'block';

      // Load sample workflow
      inputEl.value = SAMPLE_WORKFLOW;
      updateLineNumbers();

      // Run initial compile
      await compile();
    } catch (err) {
      const errDiv = document.createElement('div');
      errDiv.className = 'le-loading-error';
      const title = document.createElement('p');
      title.innerHTML = '<strong>Failed to load the WebAssembly compiler</strong>';
      const msg = document.createElement('p');
      msg.textContent = err.message || String(err);
      const hint = document.createElement('p');
      hint.style.cssText = 'margin-top:0.5rem;font-size:0.85rem;color:var(--sl-color-gray-4);';
      hint.textContent = 'Try refreshing the page. The compiler requires a modern browser with WebAssembly support.';
      errDiv.append(title, msg, hint);
      loadingEl.textContent = '';
      loadingEl.appendChild(errDiv);
    }
  }

  init();
`}</script>
